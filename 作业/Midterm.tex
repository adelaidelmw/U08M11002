\textbf{傅立叶级数的指数形式}
\begin{align*} 
    f(t) &= \sum_{n=-\infty}^{\infty} F_n e^{jn\Omega t} \\
    F_n &= \frac{1}{T} \int_{-T/2}^{T/2} f(t) e^{-jn\Omega t} dt
\end{align*}
